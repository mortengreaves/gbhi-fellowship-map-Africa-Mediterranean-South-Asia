        function getColor(region) {
            if (region === "east") return "#2c7da0";
            return "#8338ec";

        function clearAllConnections() {
            currentLines.forEach(line => africaMap.removeLayer(line));
            currentLines = [];
        }
        
        function drawConnectionsForFellow(fellowId) {
            clearAllConnections();
            const fellow = fellows[fellowId];
            if (!fellow || !fellow.connections) return;
            fellow.connections.forEach(connId => {
                const target = fellows[connId];
                if (target) {
                    const line = L.polyline([[fellow.lat, fellow.lng], [target.lat, target.lng]], {
                        color: '#2c7da0', weight: 3, opacity: 0.85
                    }).addTo(africaMap);
                    currentLines.push(line);
                }
            });
        }
        
        function showFellowDetails(fellowId) {
            const f = fellows[fellowId];
            if (!f) return;
            const regionName = f.region === "east" ? "East Africa" : "Southern Africa";
            const flagEmoji = f.country === "Ethiopia" ? "🇪🇹" : f.country === "Mauritius" ? "🇲🇺" : f.country === "Kenya" ? "🇰🇪" : f.country === "Zambia" ? "🇿🇲" : f.country === "Zimbabwe" ? "🇿🇼" : f.country === "Namibia" ? "🇳🇦" : f.country === "South Africa" ? "🇿🇦" : "🌍";
            
            const connectionsList = f.connections.map(connId => {
                const c = fellows[connId];
                if (!c) return '';
                return `<li onclick="selectConnectedFellow('${connId}')">
                            <i class="fas fa-user-friends"></i> <strong>${c.name}</strong> (${c.country})
                        </li>`;
            }).join('');
